SHOW CREATE TABLE


DROP TABLE IF EXISTS torneos;
CREATE TABLE `torneos` (
  `id` int NOT NULL AUTO_INCREMENT,
  `nombre` enum('Copa de Verano','Apertura','Copa de Invierno','Clausura') NOT NULL,
  `año` int NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
DROP TABLE IF EXISTS equipos;
CREATE TABLE `equipos` (
  `id` int NOT NULL AUTO_INCREMENT,
  `nombre` varchar(100) NOT NULL,
  `torneo_id` int NOT NULL,
  `imagen` varchar(255) DEFAULT NULL,
  `barrio` enum('Agronomía','Almagro','Balvanera','Barracas','Belgrano','Boedo','Caballito','Chacarita','Coghlan','Colegiales','Constitución','Flores','Floresta','La Boca','La Paternal','Liniers','Mataderos','Monte Castro','Montserrat','Nueva Pompeya','Núñez','Palermo','Parque Avellaneda','Parque Chacabuco','Parque Chas','Parque Patricios','Puerto Madero','Recoleta','Retiro','Saavedra','San Cristóbal','San Nicolás','San Telmo','Vélez Sársfield','Versalles','Villa Crespo','Villa del Parque','Villa Devoto','Villa General Mitre','Villa Lugano','Villa Luro','Villa Ortúzar','Villa Pueyrredón','Villa Real','Villa Riachuelo','Villa Santa Rita','Villa Soldati','Villa Urquiza','Vicente López','Olivos','Florida','La Lucila','Villa Martelli','Florida Oeste','Munro','Carapachay','Villa Adelina','San Isidro','San Fernando','Tigre','Escobar','Pilar','José C. Paz','Malvinas Argentinas','Tres de Febrero','Morón','Hurlingham','Ituzaingó','Merlo','Moreno','La Matanza','Lomas de Zamora','Lanús','Avellaneda','Quilmes','Berazategui','Florencio Varela','Ezeiza','San Martín','Villa Ballester','Villa Lynch') DEFAULT 'Palermo',
  PRIMARY KEY (`id`),
  KEY `torneo_id` (`torneo_id`),
  CONSTRAINT `equipos_ibfk_1` FOREIGN KEY (`torneo_id`) REFERENCES `torneos` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=20 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci

DROP TABLE IF EXISTS jugadores;
CREATE TABLE `jugadores` (
  `id` int NOT NULL AUTO_INCREMENT,
  `nombre` varchar(100) NOT NULL,
  `apellido` varchar(100) NOT NULL,
  `dni` varchar(20) NOT NULL,
  `dorsal` int NOT NULL,
  `goles` int DEFAULT '0',
  `amarillas` int DEFAULT '0',
  `rojas` int DEFAULT '0',
  `azules` int DEFAULT '0',
  `equipo_id` int NOT NULL,
  `fecha_nacimiento` date DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `equipo_id` (`equipo_id`),
  CONSTRAINT `jugadores_ibfk_2` FOREIGN KEY (`equipo_id`) REFERENCES `equipos` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=26 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci

DROP TABLE IF EXISTS partidos;
CREATE TABLE `partidos` (
  `id` int NOT NULL AUTO_INCREMENT,
  `torneo_id` int NOT NULL,
  `equipo_local_id` int NOT NULL,
  `equipo_visitante_id` int NOT NULL,
  `fecha` date DEFAULT NULL,
  `hora` time DEFAULT NULL,
  `goles_local` int DEFAULT '0',
  `goles_visitante` int DEFAULT '0',
  `fase` enum('Grupos','Octavos de final','Cuartos de final','Semifinal','Final') NOT NULL,
  `grupo_fecha` int DEFAULT NULL,
  `jugado` tinyint(1) DEFAULT '0',
  PRIMARY KEY (`id`),
  KEY `torneo_id` (`torneo_id`),
  KEY `equipo_local_id` (`equipo_local_id`),
  KEY `equipo_visitante_id` (`equipo_visitante_id`),
  CONSTRAINT `partidos_ibfk_4` FOREIGN KEY (`torneo_id`) REFERENCES `torneos` (`id`) ON DELETE CASCADE,
  CONSTRAINT `partidos_ibfk_5` FOREIGN KEY (`equipo_local_id`) REFERENCES `equipos` (`id`) ON DELETE CASCADE,
  CONSTRAINT `partidos_ibfk_6` FOREIGN KEY (`equipo_visitante_id`) REFERENCES `equipos` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=12 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci


DROP TABLE IF EXISTS estadisticas_partido;
CREATE TABLE `estadisticas_partido` (
  `id` int NOT NULL AUTO_INCREMENT,
  `partido_id` int NOT NULL,
  `jugador_id` int NOT NULL,
  `tipo` enum('gol','amarilla','roja','azul') NOT NULL,
  `tipo_gol` enum('jugada','penal','en_contra') DEFAULT NULL,
  `minuto` int NOT NULL,
  PRIMARY KEY (`id`),
  KEY `estadisticas_partido_ibfk_1` (`partido_id`),
  KEY `estadisticas_partido_ibfk_2` (`jugador_id`),
  CONSTRAINT `estadisticas_partido_ibfk_1` FOREIGN KEY (`partido_id`) REFERENCES `partidos` (`id`) ON DELETE CASCADE,
  CONSTRAINT `estadisticas_partido_ibfk_2` FOREIGN KEY (`jugador_id`) REFERENCES `jugadores` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=549 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci

DROP TABLE IF EXISTS inscripciones;

CREATE TABLE `inscripciones` (
  `id` int NOT NULL AUTO_INCREMENT,
  `nombre` varchar(100) DEFAULT NULL,
  `equipo` varchar(100) DEFAULT NULL,
  `telefono` varchar(20) DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci



CREATE TABLE `posiciones` (
  `id` int NOT NULL AUTO_INCREMENT,
  `equipo_id` int NOT NULL,
  `pj` int DEFAULT '0',
  `pg` int DEFAULT '0',
  `pe` int DEFAULT '0',
  `pp` int DEFAULT '0',
  `gf` int DEFAULT '0',
  `gc` int DEFAULT '0',
  `puntos` int DEFAULT '0',
  `torneo_id` int DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `torneo_equipo_unico` (`torneo_id`,`equipo_id`),
  KEY `posiciones_equipo_fk` (`equipo_id`),
  CONSTRAINT `posiciones_equipo_fk` FOREIGN KEY (`equipo_id`) REFERENCES `equipos` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=244 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci


DROP TABLE IF EXISTS noticias;
CREATE TABLE `noticias` (
  `id` int NOT NULL AUTO_INCREMENT,
  `titulo` varchar(255) NOT NULL,
  `copete` text,
  `cuerpo` text,
  `imagen_url` varchar(500) DEFAULT NULL,
  `autor` varchar(100) DEFAULT NULL,
  `fecha_publicacion` datetime DEFAULT CURRENT_TIMESTAMP,
  `publicada` tinyint(1) DEFAULT '0',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=17 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
